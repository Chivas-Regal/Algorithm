    # HDUOJ1370_Biorhythms

    # 🔗

    https://acm.dingbacode.com/showproblem.php?pid=1370

    # 💡

    问题简化成下列三个式子:
    m[1]=23, m[2]=28, m[3]=33
    x≡a[1](mod m[1])
    x≡a[2](mod m[2])
    x≡a[3](mod m[3])
    解此中国剩余定理即可
    最后求得的Ex_crt()-d要改成最小正整数

const int N = 3;
int m[N + 1] = {0, 23, 28, 33};

inline int Ex_gcd(int a, int b, int &x, int &y){
        if (b == 0) { x = 1, y = 0; return a; }
        int d = Ex_gcd(b, a % b, x, y);

        int tmp = x; x = y; y = tmp - a / b * y;

        return d;
}

inline int ksc(int a, int b, int mod){
        int res = 0;
        while(b){
                if (b & 1) res = (res + a) % mod;
                a = a * 2 % mod;
                b >>= 1;
        }return res;
}

inline int Ex_crt(){
        int X = a[1], M = m[1];
        for(int i = 2, t, y; i <= N; i ++){
                int gcd = Ex_gcd(M, m[i], t, y), mDIVgcd = m[i] / gcd;
                int c = (a[i] - X % m[i] + m[i]) % m[i];
                if(c % gcd) return -1;
                t = ksc(t, c / gcd, mDIVgcd);

                X += t * M;
                M *= mDIVgcd;
                X = (X % M + M) % M;
        }return X;
}

const int N = 100010;
ll a[N], m[N], n; // a[i]X ≡ c(mod m[i])

inline ll Ex_gcd(ll a, ll b, ll &x, ll &y){
        if(b == 0) { x = 1; y = 0; return a; }
        ll d = Ex_gcd(b, a % b, x, y);
        ll tmp = x; x = y; y = tmp - (a / b) * y;
        return d;
}

inline ll ksc(ll a, ll b, ll mod){//快速乘、好取模
        ll res = 0;
        while(b){
                if(b & 1) res = (res + a) % mod;
                a = (a + a) % mod;
                b >>= 1;
        }return res;
}

inline ll Ex_crt(){
        ll X = a[1], M = m[1];  //前一步的X，前一步的lcm
        for (ll i = 2, t, y; i <= n; i ++){
                ll gcd = Ex_gcd(M, m[i], t, y), miDIVgcd = m[i] / gcd; // 求得gcd，并使m[i]约分一下好乘进M里面
                ll c = (a[i] - X % m[i] + m[i]) % m[i];                // ax≡c(mod b) 等式右侧的c，并让他变成可行的最小正整数
                if(c % gcd) return -1;
                t = ksc(t, c / gcd, miDIVgcd);                         // 因为扩欧求得的是等号右侧为gcd时的x解，而此时等号右端为c，需要让X乘上c/gcd个t，此时先给t变了再说


                X += t * M;
                M *= miDIVgcd;        //计算LCM
                X = (X % M + M) % M;  //保持最小非负整数解
        }return X == 0? M : X; //输出最小正整数
}


CHIVAS_{
        n = inputLL();
        for(int i = 1; i <= n; i ++) m[i] = inputLL(), a[i] = inputLL();
        outLL(Ex_crt());
        _REGAL;
};
