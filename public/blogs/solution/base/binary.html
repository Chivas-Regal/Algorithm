    带环的图很难求最长环

    const int N = 110;

int n, m;
int a[N], b[N];

int dp[N][M];

inline bool Check ( int tim ) {
        memset(dp, -0x3f3f3f3f, sizeof dp);
        dp[0][0] = 0;
        for ( int i = 1; i <= n; i ++ ) {
                for ( int j = 0; j <= m; j ++ ) {         // 一共做了 j 个模块1

                }

                if ( Check(mid) ) r = mid - 1;
                else l = mid + 1;
        }
