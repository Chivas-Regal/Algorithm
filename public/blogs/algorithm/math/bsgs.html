    BSGS

    求解 at≡b(modp)gcd(a,p)=1a^t\equiv b(mod\;p)\quad gcd(a,p)=1at≡b(modp)gcd(a,p)=1

    由于欧拉定理 aϕ(p)≡1(modp)a^{\phi(p)}\equiv1(mod\;p)aϕ(p)≡1(modp)
    也就是 at≡axmodϕ(p)(modp)a^t\equiv a^{x\;mod\;\phi(p)}(mod\;p)at≡axmodϕ(p)(modp) 从而得到 t∈[0,ϕ(p)−1]t\in [0, \phi(p)-1]t∈[0,ϕ(p)−1]
    而我们在 [0,p][0,p][0,p] 里面求解，使 k=⌈p⌉k=\left\lceil\sqrt{p}\right \rceilk=⌈p​⌉
    此时 t=kx−y,{1≤x≤k,0≤y≤k−1}t=kx-y,\{1\le x\le k,\quad0\le y\le k-1\}t=kx−y,{1≤x≤k,0≤y≤k−1}
    问题转化问求 akx−y≡b(modp)a^{kx-y}\equiv b(mod\;p)akx−y≡b(modp)
    即 akx≡b⋅ay(modp)a^{kx}\equiv b·a^y(mod\;p)akx≡b⋅ay(modp)
    使用hash表存入 b⋅ay(modp)b·a^y(mod\;p)b⋅ay(modp) ,然后枚举 akxa^{kx}akx 看看hash表内有没有这个值

    #include <iostream>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <unordered_map>
#define ll long long

    using namespace std;
    

    inline int bsgs ( int a, int b, int p ) {
if ( 1 % p == b % p ) return 0;
int k = sqrt (p) + 1;
unordered_map<int, int> hash;
for ( int i = 0, j = b % p; i < k; i ++ ) {
hash[j] = i;
j = (ll)j * a % p;
}
int ak = 1;
for ( int i = 0; i < k; i ++ ) ak = (ll)ak * a % p;
for ( int i = 1, j = ak; i <= k; i ++ ) {
if ( hash.count(j) ) return (ll)i * k - hash[j];
j = (ll)j * ak % p;
}
return -1;
}
    

    int main () {
int a, p, b;
while ( cin >> a >> p >> b, a || p || b) {
int res = bsgs(a, b, p);
if ( res == -1 ) puts("No Solution");
else cout << res << endl;
}
}

    {a0≡b(modp),t=0a0!=b(modp),设gcd(a,p)=d{d=1,BSGS直接求解d>1,at≡b(modp)at+kp=b(若d%b≠0,无解返回)ad⋅at−1+kpd=bdad⋅at−1≡bd(modpd)at−1≡bd⋅ad−1(modpd){if(gcd(at−1,pd))=1BSGS直接求解else继续递归\left{\begin{aligned} &a^0\equiv b(mod;p),t=0\ &a^0;!= b(mod;p),设gcd(a,p)=d \left{\begin{aligned} &d=1,\quad BSGS直接求解\ \ &d>1,\quad\begin{aligned} &a^t\equiv b(mod;p)\ &a^t+kp=b\quad (若d% b\ne0,无解返回) \ &\frac ad·a^{t-1}+k\frac pd=\frac bd\ &\frac ad·a^{t-1}\equiv \frac bd(mod;\frac pd)\ &a^{t-1}\equiv\frac bd·\frac ad^{-1}(mod;\frac pd) \left{\begin{aligned} &if(;gcd(a^{t-1},\frac pd);)=1\quad &BSGS直接求解\ &else\quad&继续递归 \end{aligned}\right. \end{aligned} \end{aligned}\right. \end{aligned}\right. ⎩⎨⎧​​a0≡b(modp),t=0a0!=b(modp),设gcd(a,p)=d⎩⎨⎧​​d=1,BSGS直接求解d>1,​at≡b(modp)at+kp=b(若d%b=0,无解返回)da​⋅at−1+kdp​=db​da​⋅at−1≡db​(moddp​)at−1≡db​⋅da​−1(moddp​)⎩⎨⎧​​if(gcd(at−1,dp​))=1else​BSGS直接求解继续递归​​​​

    #include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <unordered_map>
#define ll long long
    

    using namespace std;
const int INF = 1e8;
    

    inline int exgcd ( int a, int b, int& x, int& y ) {
if ( !b ) { x = 1, y = 0; return a; }
int d = exgcd ( b, a % b, y, x );
y -= a / b * x;
return d;
}
    

    inline int bsgs ( int a, int b, int p ) {
if ( 1 % p == b % p ) return 0;
int k = sqrt(p) + 1;
unordered_map<int, int> hash;
for ( int i = 0, j = b % p; i < k; i ++ ) {
hash[j] = i;
j = (ll)j * a % p;
}
int ak = 1;
for ( int i = 0; i < k; i ++ ) ak = (ll)ak * a % p;
for ( int i = 1, j = ak; i <= k; i ++ ) {
if (hash.count(j)) return i * k - hash[j];
j = (ll)j * ak % p;
}
return -INF;
}
    

    inline int exbsgs ( int a, int b, int p ) {
b = ( b % p + p ) % p; // b变成正的
if ( 1 % p == b % p ) return 0;
int x, y;
int d = exgcd ( a, p, x, y );
if ( d > 1 ) { // a与p不互质，继续递归
if ( b % d ) return -INF; // 若b不是gcd的倍数
exgcd (a / d, p / d, x, y); // exgcd求逆元
return exbsgs ( a, (ll)b / d * x % p % (p / d), p / d) + 1; // 因为本来求的是t-1的最小值，+1得t
}
return bsgs(a, b, p);
    

    }
    

    int main () {
int a, p, b;
while ( cin >> a >> p >> b, a || p || b ) {
int res = exbsgs(a, b, p);
if ( res == -INF ) puts("No Solution");
else cout << res << endl;
}
}
