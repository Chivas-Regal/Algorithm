    # 🎈ksm

    基于一种分块模式的双向累乘
    将时间复杂度优化为O(logn)O(logn)O(logn)
    默认搜到此篇均会ksm，所以不细讲了，如果不会请去提前了解

    inline ll ksm(ll a, ll b, ll mod){
        ll res = 1;
        while(b){
                if(b & 1) res = res * a % mod;
                a = a * a % mod;//底数平方
                b >>= 1;//指数减半
        }return res;
}

    # 🎈矩阵乘法

    一个m×nm\times nm×n的矩阵A(m,n)A(m,n)A(m,n)左乘一个n×pn\times pn×p的矩阵B(n,p)B(n,p)B(n,p)会得到一个m×pm\times pm×p的矩阵C(m,p)C(m,p)C(m,p)
    结果是一个mmm行ppp列的矩阵，其中第iii行第jjj列位置上的数为：第一个矩阵第iii行上的nnn个数与第二个矩阵第jjj列上的nnn个数对应相乘后所得的nnn个乘积之和

    A[1][1]A[1][2]*B[1][1]B[1][2]=A[1][1]* B[1][1]+A[1][2]* B[2][1]A[1][1]* B[1][2]+A[1][2]* B[2][2]
    A[2][1]A[2][2]B[2][1]B[2][2]A[2][1]* B[1][1]+A[2][2]* B[2][1]A[1][2]* B[1][2]+A[2][2]* B[2][2]

    特点
    满足结合律和分配律，不满足交换律
    只有当矩阵AAA的列数与矩阵BBB的行数相等时，矩阵A×BA\times BA×B才有意义

    # 😄应用

    一个010101循环串，长度为L（L≤100)L（L\le100)L（L≤100)，这个擦混每秒都会进行一次变换，变换规则是：如果左边是111，则改变自己的状态，否则保持不变。给定初始状态，问nnn秒以后这个串状态
    状态转移方程：定义f(n,l)f_{(n,l)}f(n,l)​表示n秒以后，第lll个字符是000还是111。

    f(n,l)=f_{(n, l)} =f(n,l)​= f(n−1,l),f(n - 1, l - 1) = 01−f(n−1,l),f(n - 1, l - 1) != 0\begin{aligned} f_{(n - 1, l)}, & \text{f{(n - 1, l - 1)} = 0} \\ 1 - f_{(n - 1, l)}, & \text{f{(n - 1, l - 1)} != 0} \end{aligned}f(n−1,l)​,1−f(n−1,l)​,​f(n-1,l-1) = 0f(n-1,l-1) != 0​
    根据同余性质化简得： f(n,l)=f(n−1,l−1)+f(n−1,l)f_{(n, l)} = f_{(n - 1, l - 1)} + f_{(n - 1, l)}f(n,l)​=f(n−1,l−1)​+f(n−1,l)​

    //其实主功能就是一个Matrix的结构体
const int maxn = /*...*/;
const int mod = /*...*/;
struct Mat{
    ll m[maxn][maxn];//j矩阵
    Mat(int flag = 0){//初始化构造函数
        for(int i = 0; i < maxn; i ++)
            for(int j = 0; j < maxn; j ++){
                m[i][j] = flag * (i == j);
            }
    }
    inline Mat Mul(Mat a, Mat b){//矩阵乘法
        Mat res(0);
        for(int i = 0; i < maxn; i ++){
            for(int j = 0; j < maxn; j ++){
                for(int k = 0; k < maxn; k ++){
                    res.m[i][j] = (res.m[i][j] + a.m[i][k] * b.m[k][j] % mod) % mod;
                }
            }
        }
        return res;
    }
    inline Mat ksm(Mat a, ll b){//矩阵的ksm实现
        Mat res(1);
        while(b){
            if(b & 1) res = Mul(res, a);
            a = Mul(a, a);
            b >>= 1;
        }return res;
    }
};

    # 🎈题目演示

    luogu《矩阵加速（数列）模板》传送门 (opens new window)

    思路：
    一个跳系数的递推式
    建立矩阵结构体
    建立B矩阵
    建立初始矩阵
    运算得B矩阵的幂
    利用第一行得出答案

    代码：

    const int maxn = 3;
const int mod = 1e9 + 7;

struct Mat{
    ll m[maxn][maxn];
    Mat(int flag = 0){
        for(int i = 0; i < maxn; i ++)
            for(int j = 0; j < maxn; j ++){
                m[i][j] = flag * (i == j);
            }
    }
    inline Mat Mul(Mat a, Mat b){
        Mat res(0);
        for(int i = 0; i < maxn; i ++){
            for(int j = 0; j < maxn; j ++){
                for(int k = 0; k < maxn; k ++){
                    res.m[i][j] = (res.m[i][j] + a.m[i][k] * b.m[k][j] % mod) % mod;
                }
            }
        }
        return res;
    }
    inline Mat ksm(Mat a, ll b){
        Mat res(1);
        while(b){
            if(b & 1) res = Mul(res, a);
            a = Mul(a, a);
            b >>= 1;
        }return res;
    }
};

/*
a[i]       1 0 1    a[i - 1]
a[i - 1]   1 0 0    a[i - 2]
a[i - 2]   0 1 0    a[i - 3]
*/

inline void solve(){
    int n; cin >> n;
    if(n == 1 || n == 2 || n == 3){
        cout << 1 << endl;
        return;
    }

    Mat cur(0);//B矩阵
    cur.m[0][0] = 1, cur.m[0][1] = 0, cur.m[0][2] = 1;
    cur.m[1][0] = 1, cur.m[1][1] = 0, cur.m[1][2] = 0;
    cur.m[2][0] = 0, cur.m[2][1] = 1, cur.m[2][2] = 0;
    
    cur = cur.ksm(cur, n - 3);//ksm运行
    cout << (cur.m[0][0] * 1 + cur.m[0][1] * 1 + cur.m[0][2] * 1) % mod << endl;//第一行与初始的竖矩阵运行的结果
}

CHIVAS{
        int cass;
        EACH_CASE(cass){
            solve();
        }
        _REGAL;
}
