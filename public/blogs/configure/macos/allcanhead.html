    关于 MacOS 配置使用万能头文件

    Chivas-Regal

    # 查询 G++ 所在位置

    终端输入
    echo | g++ -v -x c++ -E -

    找到如下显示

    # 修改 G++ 内容

    前往上图中划线文件夹
    cd /Library/Developer/CommandLineTools/usr/bin/../include/c++/v1

    添加文件夹 bits
    mkdir bits

    并前往
    cd bits

    创建文件名 stdc++.h，并编写万能头文件代码
    vim stdc++.h
