引用时要注意一下，vuePress (opens new window)关于markdown it (opens new window)有详细介绍

只需在 config.js 中做如下配置即可

var mathjax3 = require('markdown-it-mathjax3');

module.exports = {
    extendsMarkdown: (md) => {
        md.use(mathjax3)
      },
}
